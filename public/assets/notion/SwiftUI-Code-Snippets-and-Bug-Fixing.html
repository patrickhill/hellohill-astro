JSON encode decode

Reusable code to load and update JSON files

Bundle-Decodable.swift

Bundle extension:

import Foundation

extension Bundle {
    func decode<T: Codable>(_ file: String) -> T {
        guard let url = self.url(forResource: file, withExtension: nil) else {
            fatalError("Failed to locate \(file) in bundle.")
        }

        guard let data = try? Data(contentsOf: url) else {
            fatalError("Failed to load \(file) from bundle.")
        }

        let decoder = JSONDecoder()
        let formatter = DateFormatter()
        formatter.dateFormat = "y-MM-dd"
        decoder.dateDecodingStrategy = .formatted(formatter)

        guard let loaded = try? decoder.decode(T.self, from: data) else {
            fatalError("Failed to decode \(file) from bundle.")
        }

        return loaded
    }
}

Animating Shapes

import SwiftUI

struct Trap: Shape {
    var insetAmount: CGFloat
    
    var animatableData: CGFloat {
        get {insetAmount}
        set {self.insetAmount = newValue}
    }
    
    func path(in rect: CGRect) -> Path {
        var path = Path()
        
        path.move(to: CGPoint(x:0,y:rect.maxY))
        path.addLine(to: CGPoint(x: insetAmount, y: rect.minY))
        path.addLine(to: CGPoint(x: rect.maxX - insetAmount, y: rect.minY))
        path.addLine(to: CGPoint(x: rect.maxX, y: rect.maxY))
        path.addLine(to: CGPoint(x: 0, y: rect.maxY))
        
        return path
    }
}

struct ContentView: View {
    @State private var insetAmount: CGFloat = 40.0
    
    var body: some View {
        VStack {
            Trap(insetAmount: insetAmount)
                .frame(width:300, height: 200)
                .onTapGesture {
                    withAnimation {
                        insetAmount = CGFloat.random(in: 0...100)
                    }
                }
        }
        
    }
}

struct ContentView_Previews: PreviewProvider {
    static var previews: some View {
        ContentView()
    }
}

ForEach on class array of items

import SwiftUI

struct Activity: Identifiable {
    var id = UUID()
    var title: String
    var description: String = ""
}

class Activities: ObservableObject {
    @Published var activities = [Activity]()
    
    func appendActivity(title: String, description: String) {
        self.activities.append(Activity(title:title, description:description))
    }
    
    init() {
        appendActivity(title: "Test Activity 1", description: "This is placeholder content so we can see at least one activity")
        appendActivity(title: "Test Activity 2", description: "This is placeholder content so we can see at least one activity")
    }
}

struct ContentView: View {
    @ObservedObject var activities = Activities()
    
    var body: some View {
        List {
            ForEach(activities.activities) { activity in
                Text(activity.title)
                Text(activity.description)
            }
        }
    }
}

struct ContentView_Previews: PreviewProvider {

        ContentView()
    }

Custom Modifiers


struct Title: ViewModifier {
    func body(content: Content) -> some View {
        content
            .font(.largeTitle)
            .foregroundColor(.red)
            .padding()
            .background(Color.blue)
    }
}

// allows for easier access to the modifier
// you can use .titleStyle() instead of .modifier(Title())
// it's a little cleaner
// i kinda prefer the other way because it makes it clear that you're using a custom modifier
extension View {
    func titleStyle() -> some View {
        self.modifier(Title())
    }
}

struct MyButton: View {
    var text: String
    
    var body: some View {
        Text("Hello")

Custom state binding

https://www.hackingwithswift.com/guide/ios-swiftui/2/2/key-points

**struct** ContentView: View {
    @State **var** agreedToTerms = false
    @State **var** agreedToPrivacyPolicy = false
    @State **var** agreedToEmails = false

    **var** body: some View {
        **let** agreedToAll = Binding<Bool>(
            **get**: {
                **self**.agreedToTerms && **self**.agreedToPrivacyPolicy && **self**.agreedToEmails
            },
            **set**: {
                **self**.agreedToTerms = $0
                **self**.agreedToPrivacyPolicy = $0
                **self**.agreedToEmails = $0
            }
        )

        **return** VStack {
            Toggle(isOn: $agreedToTerms) {
                Text("Agree to terms")
            }

            Toggle(isOn: $agreedToPrivacyPolicy) {
                Text("Agree to privacy policy")
            }

            Toggle(isOn: $agreedToEmails) {
                Text("Agree to receive shipping emails")
            }

            Toggle(isOn: agreedToAll) {
                Text("Agree to all")
            }
        }
    }
}
